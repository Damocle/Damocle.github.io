getElementsByClassName的兼容代码 

// getElementsByClassName() 返回文档中所有指定类名的元素集合，作为NodeList对象
// element 包括doucument或父节点parenetElement
// names 目标元素的所有class类名
function getElementsByClassName(element, names) {
    if (element.getElementsByClassName) { // 若浏览器支持该方法
        return element.getElementsByClassName(names);
    } else {                              // 不支持
        // 获取element的所有子元素节点
        var elements = element.getElementsByTagName("*"),
            result   = [];   // 存储最终获取的目标元素集合
        var element,
            classNameStr, 
            flag;

        names = names.split(" "); // 获得包含所有class类名的数组
        // 循环父元素中的所有子元素节点
        for (var i = 0; element = elements[i]; i++) {
            classNameStr = " " + element.className + " "; // 存储子元素节点的class类名
            flag = true;
            // 判断该子元素节点的class类名是否与所求目标元素的class类名相等
            for (var j = 0, name; name = names[j]; j++) {
                // 若该子节点元素不符合则排除该子节点元素跳出当前循环
                if (classNameStr.indexOf(" " + name + " ") == -1) {
                    flag = false; // 排除该子节点元素
                    break;
                }
            };

            // 该子节点元素的class类名与目标节点元素class类名相符则将该元素存入result数组
            if (flag) {
                result.push(element); 
            }
        }

        return result;  // 返回结果集
    }
}
