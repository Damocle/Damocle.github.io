日期级联下拉选择框代码实现

需求：

1、“年”或“月”发生变化且为有效值时，“日”下拉选择框显示相应的项。

2、“年”下拉选择框为初始状态（即值为0）时，“月”、“日”下拉选择框一定为初始状态（即值为0）

3、“月”下拉选择框为初始状态（即值为0）时， “日”下拉选择框一定为初始状态（即值为0）。

代码实现：

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8" />
<title>日期级联下拉选择框代码实现</title>
<script>
window.onload = function() {
var optionYear  = document.getElementById("year");    
var  optionMonth = document.getElementById("Month");  
var optionDay   = document.getElementById("date");

// 获取每年每月的相应的天数
function getDay(year, month) {
    var day = 0,
       isLeapYear = false;
    
    if (year != 0 && month != 0) {
        // 判断是否为闰年
        if (month == 2) {
            var num = year % 100;
            if (num == 0) {
                num = year % 400;
                if (num == 0) {
                  isLeapYear = true;
                }
            } else {
                num = year % 4;
                if(num == 0) {
                    isLeapYear = true;
                }
            }
            
            if (isLeapYear) {
                day = 29;
            } else {
                    day = 28;
            }
          }
          
          switch(month) {
              case "1":
              case "3":
              case "5":
              case "7":
              case "8":
              case "10":
              case "12":
                  day = 31;
                  break;
              case "2":
              case "4":
              case "6":
              case "9":
              case "11":
                  day = 30;
                  break;
              
          }
          
          for (var i = 1; i <= day; i++) {
              optionDay.options.add(new Option(i, i))
          }
        
    }

}

// “年”下拉选择框为初始状态（即值为0）时，“月”、“日”下拉选择框一定为初始状态（即值为0）
function stateYear() {
    if (optionYear.value == 0) {
        optionMonth.value = 0;
        optionDay.value = 0;
    }
} 
// “月”下拉选择框为初始状态（即值为0）时， “日”下拉选择框一定为初始状态（即值为0）
function stateMonth() {
    if (optionMonth.value == 0) {
        optionDay.value = 0;
    }
} 

optionYear.onclick = function() {
    stateYear();
}
optionMonth.onclick = function() {
    stateYear();
    stateMonth();
}
optionDay.onmouseover = function() {
    stateYear();
    stateMonth();
    getDay(optionYear.value, optionMonth.value);
}
}
</script>   
</head>

</select>年

    <option value="9">9</option>

    <option value="12">12</option>
</select>月
<select name="date" id="date">
    <option value="0">--</option>
</select>日
</body>
