最近在完成一个作业时，需要用到一个评论功能，首先想到的是用php、ajax，之后思考了一下大概的制作流程如下：

  1. 前台显示页面设计、制作（html、css）
  2. javascript获取表单数据（省去验证），利用Ajax-post方法发送数据给后台
  3. 后台(html代码省略)php接收Ajax处发送的数据、验证并插入数据表（数据库表设计）
  4. 插入成功则直接返回值给Ajax
  5. Ajax创建显示评论数据的节点，并插入html页面
  6. 前台php查询数据表展示所有评论信息

    // getElementsByClassName 函数
function getElementsByClassName(str, root, tag) {
     if (root) {
         root = typeof root == "string" ? document.getElementById(root) : root;
     } else {
         root = document.body;
     }

     tag = tag || "*";
     var els = root.getElementsByTagName(tag),
     arr = [];

     for (var i = 0, n = els.length; i < n; i++) {
         for (var j = 0, k = els[i].className.split(" "), l = k.length; j < l; j++) {
             if (k[j] == str) {
                 arr.push(els[i]);
                 break;
             }
         }
     }

     return arr;
}

/* post函数参数： 
     url: 请求地址
     requestData： 发送的数据
     func: 获取文件的处理函数
*/
function post(url, requestData, func) {
     var xhr = createRequest();
     xhr.open("post", url, true);
     xhr.onreadystatechange = function (){
         if (xhr.readyState == 4) {
             if ((xhr.status >= 200 && xhr.status < 300) || xhr.status == 304) {
                 func(xhr.responseText);
             } else {
                 alert("request failed : " + xhr.status);
             }
         }
     }
     xhr.setRequestHeader("Content-type","application/x-www-form-urlencoded; charset=utf-8");
     xhr.send(requestData);
 }
